/**
 * Owlka waitlist webhook — Google Apps Script.
 *
 * Writes a row to the active sheet for each POSTed signup. Column
 * order is:
 *   A  timestamp   (ISO8601 from the website, falls back to now)
 *   B  email
 *   C  firstName
 *   D  lastName
 *   E  source      (e.g. "owlka.com")
 *
 * Deploy instructions:
 *  1. Open the target Google Sheet. Row 1 headers should match the
 *     column order above.
 *  2. Extensions -> Apps Script. Replace the default Code.gs contents
 *     with this file, save (Cmd+S).
 *  3. Deploy -> Manage deployments -> pencil on the existing
 *     deployment -> Version: New version -> Deploy. The web app URL
 *     does not change.
 *  4. If this is the first deployment instead: Deploy -> New
 *     deployment -> Type: Web app. Execute as: Me. Who has access:
 *     Anyone. Click Deploy, authorise, copy the Web app URL.
 *  5. Paste the URL into Vercel as env var WAITLIST_SHEET_WEBHOOK
 *     (Production + Preview) and redeploy owlka-website.
 */
function doPost(e) {
  const payload = JSON.parse(e.postData.contents);
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  sheet.appendRow([
    payload.timestamp || new Date().toISOString(),
    payload.email || "",
    payload.firstName || "",
    payload.lastName || "",
    payload.source || "",
  ]);
  return ContentService.createTextOutput(
    JSON.stringify({ ok: true })
  ).setMimeType(ContentService.MimeType.JSON);
}
